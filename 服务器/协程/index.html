int ret = poll(fd);

recv(fd,buffer,length,0);
}

int async_recv(fd,buffer,length,0){

int ret = poll(fd);
if (ret == -1) 
    swap(); //切换下一个调度体  如何回到上一个fd?

recv(fd,buffer,length,0); //并没有由 recv 操作 io

yeild(); //主动让出cpu

}

swap 需要做到 线程切换

CPU 寄存器里面有保存当前运行的代码
我们可以先保存当前的线程上下文
之后切换到别的 线程上下文继续执行

至于切换到哪一个协程，或者让出哪一个cpu 都需要调度器来协助。
至于协程切换的代价就和函数调用一样。
